version: "3.8"

services:
  redpanda-console:
    image: docker.redpanda.com/redpandadata/console:latest
    container_name: redpanda-console
    ports:
      - "8080:8080"
    environment:
      KAFKA_BROKERS: "kafka:9092"     # адрес твоего Kafka-брокера
      CONSOLE_LISTEN_ADDRESS: "0.0.0.0:8080"

Заходим в браузере на IP:8080

services:
  kafka-ui:
    image: provectuslabs/kafka-ui:latest
    container_name: kafka-ui
    ports:
      - "8080:8080"
    environment:
      KAFKA_CLUSTERS_0_NAME: "local"
      KAFKA_CLUSTERS_0_BOOTSTRAPSERVERS: "kafka:9092" # замени на адрес твоего брокера

Заходим в браузере на IP:8080
